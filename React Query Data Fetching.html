import { useQuery } from '@tanstack/react-query'; import api from '../api/axiosInstance'; // Axios instance with auth interceptors const fetchUserProfile = async () => { const { data } = await api.get('/api/users/profile'); return data; }; export const useUser = () => { return useQuery({ queryKey: ['userProfile'], queryFn: fetchUserProfile, staleTime: 1000 * 60 * 5, // Cache for 5 minutes }); };
